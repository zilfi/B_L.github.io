bl.compute.risk.aversion <- function(r_mrk, risk.free = 0)
{
  lambda = mean(coredata(r_mrk) - coredata(risk.free)) / var(coredata(r_mrk))
  return( as.double(lambda) )
}

bl.compute.posterior <- function (equilibruim.return, ret_matrix, pmat=NULL, qmat=NULL, tau=0.025)
  {
  om<-tau * pmat %*%cov(ret_matrix)%*% t(pmat)
  omega=diag(diag(om), nrow=nrow(om),ncol=ncol(om))
  #(1)
  firstform<-solve(solve(tau*cov(ret_matrix))+ t(pmat)%*%solve(omega)%*%pmat)
  #(2)
  secondform<-(solve(tau*cov(ret_matrix))%*%equilibruim.return + t(pmat)%*%solve(omega)%*%qmat)
    
  final<-firstform%*%secondform
  final=as.vector(final)
  names(final)<-paste(watchlist[c(-(length(data[1,])-1),-length(data[1,]))], "IMPLIED")
  return(final)
}

bl.compute.optimal <- function(risk.aversion, bl.posterior, cov)
{
  return( (1/risk.aversion) * solve(cov) %*% mu )
}
